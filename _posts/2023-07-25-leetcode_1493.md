# 내풀이

```js
/*
solution:
저번 문제에서 k를 1로 바꿔서 푸는 문제

time:O(N)
space:O(1)
*/

var longestSubarray = function(nums) {
  let l = 0;
  let r = 0;
  let k = 1;
  let max = 0;
  
  while(r < nums.length){
    if(nums[r] == 0 && k)k--;
    else if(nums[r] == 0 && !k){
      while(nums[l] != 0)l++;
      l++;
    }
    max = Math.max(max, r - l);
    r++;
  }
  return max;
};
```
